# creating a function to obtain the value of the outlier 
# Lowerbound 
Q1 = df['Age'].quantile(0.25)
# Upperbound
Q3 = df['Age'].quantile(0.75)
# Interquantile range
IQR = Q3-Q1
def outliers (df):
    outliers_r = df[((df<(Q1-1.5*IQR)) | (df>(Q3+1.5*IQR)))]
    return outliers_r
